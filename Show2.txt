[
  {
    title: "Bill Gates",
    author: "The Road Ahead",
    readingStatus: true,
  },
  {
    title: "Steve Jobs",
    author: "Walter Isaacson",
    readingStatus: true,
  }
]

result => 
[
  [
    'title: Bill Gates',
    'author: The Road Ahead',
    'readingStatus: true'
  ],
  [
    'title: Steve Jobs',
    'author: Walter Isaacson',
